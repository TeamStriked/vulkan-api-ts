VkPerformanceValueTypeINTEL = {
  VK_PERFORMANCE_VALUE_TYPE_UINT32_INTEL: 0,
  VK_PERFORMANCE_VALUE_TYPE_UINT64_INTEL: 1,
  VK_PERFORMANCE_VALUE_TYPE_FLOAT_INTEL: 2,
  VK_PERFORMANCE_VALUE_TYPE_BOOL_INTEL: 3,
  VK_PERFORMANCE_VALUE_TYPE_STRING_INTEL: 4
};
 Properties VkPerformanceValueTypeINTEL.VK_PERFORMANCE_VALUE_TYPE_UINT32_INTEL specifies that unsigned 32-bit integer data is returned in data.value32. VkPerformanceValueTypeINTEL.VK_PERFORMANCE_VALUE_TYPE_UINT64_INTEL specifies that unsigned 64-bit integer data is returned in data.value64. VkPerformanceValueTypeINTEL.VK_PERFORMANCE_VALUE_TYPE_FLOAT_INTEL specifies that floating-point data is returned in data.valueFloat. VkPerformanceValueTypeINTEL.VK_PERFORMANCE_VALUE_TYPE_BOOL_INTEL specifies that Bool32 data is returned in data.valueBool. VkPerformanceValueTypeINTEL.VK_PERFORMANCE_VALUE_TYPE_STRING_INTEL specifies that a reference to a null-terminated string is returned in data.valueString. The reference is valid for the lifetime of the device parameter passed to vkGetPerformanceParameterINTEL.
