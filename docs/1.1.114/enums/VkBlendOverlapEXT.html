VkBlendOverlapEXT = {
  VK_BLEND_OVERLAP_UNCORRELATED_EXT: 0,
  VK_BLEND_OVERLAP_DISJOINT_EXT: 1,
  VK_BLEND_OVERLAP_CONJOINT_EXT: 2
};
 Properties VkBlendOverlapEXT.VK_BLEND_OVERLAP_UNCORRELATED_EXT specifies that there is no correlation between the source and destination coverage. VkBlendOverlapEXT.VK_BLEND_OVERLAP_DISJOINT_EXT specifies that the source and destination coverage are considered to have minimal overlap. VkBlendOverlapEXT.VK_BLEND_OVERLAP_CONJOINT_EXT specifies that the source and destination coverage are considered to have maximal overlap.
